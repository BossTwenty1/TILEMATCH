Single React Artifact | No Build Tools | No localStorage | No External Router

4. State Architecture
Three Context providers wrap the entire app. No data is written to localStorage.
AuthContext
js{
  currentUser: null | userObject,
  isAdmin: false | true,
  login(email, password): boolean,
  logout(): void,
  register(formData): boolean,
  users: [...mockUsers]
}
CartContext
js{
  items: [{ product, qty }],
  addItem(product, qty): void,
  removeItem(productId): void,
  updateQty(productId, qty): void,
  clearCart(): void,
  itemCount: number,
  subtotal: number
}
StoreContext
js{
  products: [...],
  orders: [...],
  addProduct(product): void,
  updateProduct(id, data): void,
  deleteProduct(id): void,
  updateStock(productId, qty): void,
  placeOrder(orderData): string,       // returns new order ID
  updateOrderStatus(orderId, status): void
}

CSV Export
jsconst exportOrdersCSV = (orders) => {
  const headers = ['Order ID', 'Customer', 'Email', 'Total', 'Status', 'Date'];
  const rows = orders.map(o =>
    [o.id, o.customerName, o.customerEmail, o.total, o.status, o.createdAt].join(',')
  );
  const csv = [headers.join(','), ...rows].join('\n');
  const blob = new Blob([csv], { type: 'text/csv' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'tilematch_orders.csv';
  a.click();
  URL.revokeObjectURL(url);
};
Email Validator
jsconst isValidEmail = (email) => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
VAT + Shipping Calculator
jsconst TAX_RATE = 0.12;
const SHIPPING_FEE = 200;
const FREE_SHIPPING_THRESHOLD = 2000;

const calculateTotals = (subtotal) => {
  const shipping = subtotal >= FREE_SHIPPING_THRESHOLD ? 0 : SHIPPING_FEE;
  const tax = subtotal * TAX_RATE;
  const total = subtotal + shipping + tax;
  return { shipping, tax, total };
};
